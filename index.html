// Questa funzione controllerà se l'EAN esiste già e precompilerà il form function checkExistingEan(barcodeText) { if (!barcodeText) return; // Cerca se esiste già un prodotto con lo stesso codice EAN const existingProduct = products.find(p => p.barcode === barcodeText); if (existingProduct) { // Precompila i dati esistenti document.getElementById('form-name').value = existingProduct.name || ''; document.getElementById('form-brand').value = existingProduct.brand || ''; document.getElementById('form-category').value = existingProduct.category || 'Dispensa'; document.getElementById('form-notes').value = existingProduct.notes || ''; // Recupera la vecchia foto se presente if (existingProduct.photo) { document.getElementById('form-photo').value = existingProduct.photo; const preview = document.getElementById('product-preview'); preview.src = existingProduct.photo; preview.classList.remove('hidden'); document.getElementById('photo-placeholder').classList.add('hidden'); } showToast("Prodotto già conosciuto! Inserisci solo la nuova scadenza."); // Porta il focus direttamente sulla data di scadenza setTimeout(() => { document.getElementById('form-expiry').focus(); }, 300); } } // Modifichiamo la funzione di scansione per integrare il controllo function startScanning() { if (!html5QrScanner) html5QrScanner = new Html5Qrcode("scanner-reader"); html5QrScanner.start({ facingMode: "environment" }, { fps: 15, qrbox: 200 }, (decodedText) => { stopScanning(); resetForm(); // Pulisce il form da residui di modifiche precedenti document.getElementById('form-barcode').value = decodedText; // CONTROLLO EAN ESISTENTE checkExistingEan(decodedText); switchView('add'); }, () => {}); }
